GoreBox Modding API

Полная документация по созданию модов с помощью Lua скриптов. Создавайте свои игровые механики, предметы и многое другое.

Начать изучение Читать блог

Быстрый старт

Создание первого скрипта

Начните с создания простого Lua модуля для управления объектами в игре.

local M = {}

function M.start()
    log("Мой первый мод загружен!")
end

return M

Работа с объектами

Используйте функции для поиска и изменения игровых объектов.

local playerID = getLocalPlayer()
if playerID then
    local t = gameObject.getTransform(playerID)
    transform:setPosition(t, 0, 10, 0)
end

Последние новости

Жизненный цикл Lua-скрипта

awake()

Инициализация Lua-среды перед запуском.

function M.awake()
    log("Module is awake!")
end

start()

Вызывается при старте скрипта.

function M.start()
    log("Module has started!")
end

update()

Обновление каждый кадр.

function M.update()
    --[[comment:game_logic]]
end

late_update()

Выполняется после update().

function M.late_update()
    --[[comment:camera_logic]]
end

fixed_update()

Срабатывает с фиксированным интервалом (для физики).

function M.fixed_update()
    --[[comment:physics_logic]]
end

gui_update()

Используется для отрисовки GUI.

function M.gui_update()
    gui.label("Hello from GUI!")
end

onRPC(message)

Обрабатывает сетевые сообщения.

function M.onRPC(message)
    if message == "Heal" then
        log("Received Heal RPC")
    end
end

Основные функции

log(message)

Выводит текст в консоль Unity.

log("Mod started!")

getModRoot()

Возвращает путь к папке мода.

local rootPath = getModRoot()
log("Mod path: " .. rootPath)

getDeltaTime()

Возвращает время между кадрами.

local dt = getDeltaTime()
transform:Translate(0, 0, 5 * dt)

time.getTime()

Возвращает время с начала игры.

local gameTime = time.getTime()
log("Game running for: " .. gameTime .. "s")

getLocalPlayer()

Возвращает объект (ID) игрока.

local playerID = getLocalPlayer()
local playerTransform = gameObject.getTransform(playerID)

Функции GUI

gui.label(text)

Отображает текст.

gui.label("Player Health: 100")

gui.space(pixels)

Отступ между элементами.

gui.label("First line")
gui.space(10)
gui.label("Second line")

gui.box(textureId, ...)

Отображает изображение (текстуру).

local myTexture = texture.load("images/icon.png")
gui.box(myTexture, true)

gui.button(text)

Кнопка, возвращает true, если нажата.

if gui.button("Click Me!") then
    log("Clicked!")
end

gui.textField(text)

Поле ввода, возвращает измененный текст.

local inputText = "Hello"
inputText = gui.textField(inputText)

gui.verticalSlider(value, min, max)

Вертикальный слайдер.

local vSlider = 50
vSlider = gui.verticalSlider(vSlider, 0, 100)

gui.horizontalSlider(value, min, max)

Горизонтальный слайдер.

local hSlider = 0.5
hSlider = gui.horizontalSlider(hSlider, 0.0, 1.0)

gui.beginVertical() / gui.endVertical()

Вертикальная группа элементов.

gui.beginVertical()
gui.label("Item 1")
gui.label("Item 2")
gui.endVertical()

gui.beginHorizontal() / gui.endHorizontal()

Горизонтальная группа элементов.

gui.beginHorizontal()
gui.label("Left")
gui.label("Right")
gui.endHorizontal()

Ресурсы

texture.load(path)

Загружает текстуру (возвращает ID).

local texID = texture.load("images/my_texture.png")

material.create(texturePath)

Создает материал с текстурой (возвращает ID).

local matID = material.create("images/my_texture.png")

material.createWithShader(texturePath, shaderName)

Создает материал с шейдером (возвращает ID).

local glowMat = material.createWithShader("tex/glow.png", "Legacy Shaders/Additive")

material.setColor(id, prop, r, g, b, a)

Изменяет цвет материала.

--[[comment:set_red]]
material.setColor(matID, "_Color", 1, 0, 0, 1)

material.getColor(id, prop)

Возвращает цвет (r, g, b, a).

local r, g, b, a = material.getColor(matID, "_Color")
log("Red: " .. r)

model.load(path)

Загружает 3D-модель (возвращает ID).

local modelID = model.load("models/my_model.obj")

gameObject.createWithModel(name, model, texture)

Создает объект с моделью и текстурой (возвращает ID).

local obj = gameObject.createWithModel("MyObject", "models/car.obj", "tex/car.png")

audioClip.load(path)

Загружает аудио (возвращает ID).

local soundID = audioClip.load("sounds/shot.wav")

Объекты

gameObject.create(name)

Создает новый пустой объект (возвращает ID).

local newObj = gameObject.create("EmptyHolder")

gameObject.find(name)

Ищет объект по имени (возвращает ID).

local player = gameObject.find("MainPlayer")

gameObject.destroy(id)

Удаляет объект.

gameObject.destroy(newObj)

gameObject.getTransform(id)

Возвращает Transform (ID) объекта.

local player = gameObject.find("MainPlayer")
local t = gameObject.getTransform(player)

gameObject.setName(id, name)

Задает имя объекту.

gameObject.setName(player, "RenamedPlayer")

gameObject.setActive(id, bool)

Активирует или деактивирует объект.

--[[comment:hide_player]]
gameObject.setActive(player, false)

gameObject.instantiate(id)

Клонирует объект (возвращает ID клона).

local playerClone = gameObject.instantiate(player)

Трансформы

transform.getPosition(id)

Возвращает позицию (x, y, z).

local x, y, z = transform.getPosition(t)
log("Y position: " .. y)

transform.setPosition(id, x, y, z)

Устанавливает позицию.

--[[comment:teleport_roof]]
transform.setPosition(t, 10, 50, 20)

transform.translate(id, x, y, z)

Сдвигает объект (локально).

--[[comment:move_forward]]
transform.translate(t, 0, 0, 1 * getDeltaTime())

transform.rotate(id, x, y, z)

Вращает объект.

--[[comment:rotate_y]]
transform.rotate(t, 0, 45 * getDeltaTime(), 0)

transform.lookAt(id, x, y, z)

Направляет объект на точку.

--[[comment:look_center]]
transform.lookAt(t, 0, 0, 0)

transform.getForward(id)

Возвращает направление "вперёд" (x, y, z).

local fwdX, fwdY, fwdZ = transform.getForward(t)

Физика

rigidbody.setVelocity(id, x, y, z)

Устанавливает скорость (ID объекта, не Rigidbody).

--[[comment:jump]]
rigidbody.setVelocity(player, 0, 10, 0)

rigidbody.addForce(id, x, y, z)

Прикладывает силу к объекту.

--[[comment:push_forward]]
rigidbody.addForce(player, 0, 0, 100)

rigidbody.setUseGravity(id, bool)

Включает/выключает гравитацию.

--[[comment:fly_mode]]
rigidbody.setUseGravity(player, false)

physics.raycast(x, y, z, dirX, dirY, dirZ)

Пускает луч, возвращает bool (попал или нет).

local didHit = physics.raycast(0, 1, 0, 0, -1, 0)
if didHit then log("Hit ground") end

physics.overlapSphere(x, y, z, r)

Возвращает таблицу ID объектов в сфере.

local targets = physics.overlapSphere(0, 0, 0, 10)
for _, id in pairs(targets) do
    log("Found: " .. id)
end

Аудио

audioSource.play(id)

Воспроизводит звук (ID объекта с AudioSource).

audioSource.play(player)

audioSource.stop(id)

Останавливает звук.

audioSource.stop(player)

audioSource.setVolume(id, v)

Устанавливает громкость (0.0 - 1.0).

audioSource.setVolume(player, 0.5)

audioSource.isPlaying(id)

Проверяет, воспроизводится ли звук.

if audioSource.isPlaying(player) then
    log("Music is playing")
end

Камера

camera.setFieldOfView(id, fov)

Устанавливает угол обзора (FOV).

local cam = gameObject.find("MainCamera")
camera.setFieldOfView(cam, 90)

camera.worldToScreenPoint(id, x, y, z)

Мировые → экранные координаты (возвращает x, y, z).

local x, y, z = camera.worldToScreenPoint(cam, 0, 0, 0)
log("Center on screen: " .. x)

camera.screenToWorldPoint(id, x, y, z)

Экранные → мировые координаты (возвращает x, y, z).

--[[comment:z_depth]]
local worldX, _, _ = camera.screenToWorldPoint(cam, 100, 100, 10)

Ввод

input.getKey(name)

Проверка, зажата ли клавиша (bool).

if input.getKey("Space") then
    log("Space is held down")
end

input.getKeyDown(name)

Проверка нажатия клавиши (bool, срабатывает 1 раз).

if input.getKeyDown("F") then
    log("Toggled flashlight")
end

input.getMouse(button)

Нажатие кнопки мыши (0=LMB, 1=RMB, 2=MMB).

if input.getMouse(0) then
    log("Left mouse button pressed")
end

input.getAxis(name)

Возвращает значение оси (напр., "Mouse X", "Vertical").

local v = input.getAxis("Vertical")
if v > 0 then log("Moving forward") end

input.setMouseLock(bool)

Управление блокировкой курсора.

--[[comment:show_cursor]]
input.setMouseLock(false)
